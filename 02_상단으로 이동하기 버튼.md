- js

```js
const toTopEl = document.querySelector('.totop')

// 스크롤위치에 따라 top버튼 숨기기
window.addEventListener('scroll', _.throttle(function () {
  if (window.scrollY > 500) {
    gsap.to(toTopEl, .2, {
      x: 0
    })
  } else {
    gsap.to(toTopEl, .2, {
      x: 100
    })
  }
}, 300))

// top버튼 클릭시 최상단으로 이동
toTopEl.addEventListener('click', function(){
  gsap.to(window, .7, {
    scrollTo: 0
  })
})
```
